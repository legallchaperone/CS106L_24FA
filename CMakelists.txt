cmake_minimum_required(VERSION 3.10)

# Set the project name
project(CS106L_Assignments)

# Specify the C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Add the assignments directory
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)
add_subdirectory(cs106l-assignments)

# Include directories
include_directories(${PROJECT_SOURCE_DIR}/cs106l-assignments)

# Add executables for each assignment
file(GLOB ASSIGNMENT_SOURCES "cs106l-assignments/*.cpp")
foreach(SOURCE ${ASSIGNMENT_SOURCES})
    get_filename_component(EXE_NAME ${SOURCE} NAME_WE)
    add_executable(${EXE_NAME} ${SOURCE})
endforeach()
